 36 (function () {
 37 	var initializing = false;
 38 	var fnTest = (/xyz/).test(function(){xyz;}) ? (/\b_super\b/) : /.*/;
 39 	
 40 	/**
 41 	 * <p>Defines a base class from which to create new classes that can be
 42 	 * extended into new classes as well.</p>
 43 	 * 
 44 	 * <p>This constructor does nothing.</p>
 45 	 * 
 46 	 * <p>Modified from: http://ejohn.org/blog/simple-javascript-inheritance/</p>
 47 	 * 
 48 	 * @class
 49 	 * @name Class
 50 	 */
 51 	this.Class = function () {};
 52 	
 53 	/**
 54 	 * <p>Creates a new class that inherits from the calling class.</p>
 55 	 * 
 56 	 * @function
 57 	 * 
 58 	 * @param {string, object} classNameOrProps
 59 	 *     Either the name of the class, or an object containing the class' properties.
 60 	 * @param {object} props
 61 	 *     An object containing the class' properties (only if classNameOrProps
 62 	 *     specifies the class name).
 63 	 */
 64 	Class.extend = function (classNameOrProps, props) {
 65 		var _super = this.prototype;
 66 		var className = props ? classNameOrProps : "Class";
 67 		props = props || classNameOrProps;
 68 		
 69 		// Instantiate a base class (but only create the instance,
 70 		// don't run the init constructor)
 71 		initializing = true;
 72 		var prototype = new this();
 73 		initializing = false;
 74 		
 75 		// Copy the properties over onto the new prototype
 76 		for (var name in props) {
 77 			// Check if we're overwriting an existing function
 78 			if (typeof props[name] == "function" &&
 79 				typeof _super[name] == "function" &&
 80 				fnTest.test(props[name])) {
 81 				prototype[name] = (function(name, fn){
 82 					return function() {
 83 					var tmp = this._super;
 84 					
 85 					// Add a new ._super() method that is the same method
 86 					// but on the super-class
 87 					this._super = _super[name];
 88 					
 89 					// The method only need to be bound temporarily, so we
 90 					// remove it when we're done executing
 91 					var ret = fn.apply(this, arguments);        
 92 					this._super = tmp;
 93 					
 94 					return ret;
 95 					};
 96 				})(name, props[name]);
 97 			} else {
 98 				prototype[name] = props[name];
 99 			}
100 		}
101 		
102 		// The new class
103 		var NewClass = function () {
104 			// All construction is actually done in the init method
105 			if (!initializing && this.init)
106 				this.init.apply(this, arguments);
107 		}
108 		
109 		NewClass.className = className;
110 		NewClass.prototype = prototype;
111 		NewClass.constructor = Class;
112 		NewClass.extend = arguments.callee;
113 		
114 		return NewClass;
115 	};
116 })();

122 /**
123  * <p>Formats the string 'format' with the given arguments. Uses the
124  * php formatting style, defined at
125  * http://php.net/manual/en/function.sprintf.php.</p>
126  * 
127  * <p>From: http://phpjs.org/functions/sprintf:522.</p>
128  * 
129  * @param {string} format
130  *     The string to be formatted with the given arguments.
131  * @param {[mixed]} rest
132  *     The rest of the arguments with which to format the given string.
133  * 
134  * @example
135  *     >> sprintf("%01.2f", 123.1)
136  *     => 123.10
137  * @example
138  *     >> sprintf("[%10s]", 'monkey')
139  *     => "[    monkey]"
140  * @example
141  *     >> sprintf("[%'#10s]", 'monkey')
142  *     => "[####monkey]"
143  */
144 function sprintf (format /*, ... */) {
145     // http://kevin.vanzonneveld.net
146     // +   original by: Ash Searle (http://hexmen.com/blog/)
147     // + namespaced by: Michael White (http://getsprink.com)
148     // +    tweaked by: Jack
149     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
150     // +      input by: Paulo Ricardo F. Santos
151     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
152     // +      input by: Brett Zamir (http://brett-zamir.me)
153     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
154 
155     var regex = /%%|%(\d+\$)?([-+\'#0 ]*)(\*\d+\$|\*|\d+)?(\.(\*\d+\$|\*|\d+))?([lscboxXuidfegEG])/g;
156     var a = arguments, i = 0, format = a[i++];
157 	
158     var pad = function (str, len, chr, leftJustify) {
159         if (!chr) {chr = ' ';}
160         var padding = (str.length >= len) ? '' : Array(1 + len - str.length >>> 0).join(chr);
161         return leftJustify ? str + padding : padding + str;
162     };
163 	
164     var justify = function (value, prefix, leftJustify, minWidth, zeroPad, customPadChar) {
165         var diff = minWidth - value.length;
166         if (diff > 0) {
167             if (leftJustify || !zeroPad) {
168                 value = pad(value, minWidth, customPadChar, leftJustify);
169             } else {
170                 value = value.slice(0, prefix.length) + pad('', diff, '0', true) + value.slice(prefix.length);
171             }
172         }
173         return value;
174     };
175 	
176     var formatBaseX = function (value, base, prefix, leftJustify, minWidth, precision, zeroPad) {
177         // Note: casts negative numbers to positive ones
178         var number = value >>> 0;
179         prefix = prefix && number && {'2': '0b', '8': '0', '16': '0x'}[base] || '';
180         value = prefix + pad(number.toString(base), precision || 0, '0', false);
181         return justify(value, prefix, leftJustify, minWidth, zeroPad);
182     };
183 
184     // formatString()
185     var formatString = function (value, leftJustify, minWidth, precision, zeroPad, customPadChar) {
186         if (precision != null) {
187             value = value.slice(0, precision);
188         }
189         return justify(value, '', leftJustify, minWidth, zeroPad, customPadChar);
190     };
191 
192     // doFormat()
193     var doFormat = function (substring, valueIndex, flags, minWidth, _, precision, type) {
194         var number;
195         var prefix;
196         var method;
197         var textTransform;
198         var value;
199 
200         if (substring == '%%') {return '%';}
201 
202         // parse flags
203         var leftJustify = false, positivePrefix = '', zeroPad = false, prefixBaseX = false, customPadChar = ' ';
204         var flagsl = flags.length;
205         for (var j = 0; flags && j < flagsl; j++) {
206             switch (flags.charAt(j)) {
207                 case ' ': positivePrefix = ' '; break;
208                 case '+': positivePrefix = '+'; break;
209                 case '-': leftJustify = true; break;
210                 case "'": customPadChar = flags.charAt(j+1); break;
211                 case '0': zeroPad = true; break;
212                 case '#': prefixBaseX = true; break;
213             }
214         }
215 
216         // parameters may be null, undefined, empty-string or real valued
217         // we want to ignore null, undefined and empty-string values
218         if (!minWidth) {
219             minWidth = 0;
220         } else if (minWidth == '*') {
221             minWidth = +a[i++];
222         } else if (minWidth.charAt(0) == '*') {
223             minWidth = +a[minWidth.slice(1, -1)];
224         } else {
225             minWidth = +minWidth;
226         }
227 
228         // Note: undocumented perl feature:
229         if (minWidth < 0) {
230             minWidth = -minWidth;
231             leftJustify = true;
232         }
233 
234         if (!isFinite(minWidth)) {
235             throw new Error('sprintf: (minimum-)width must be finite');
236         }
237 
238         if (!precision) {
239             precision = 'fFeE'.indexOf(type) > -1 ? 6 : (type == 'd') ? 0 : undefined;
240         } else if (precision == '*') {
241             precision = +a[i++];
242         } else if (precision.charAt(0) == '*') {
243             precision = +a[precision.slice(1, -1)];
244         } else {
245             precision = +precision;
246         }
247 		
248         // grab value using valueIndex if required?
249         value = valueIndex ? a[valueIndex.slice(0, -1)] : a[i++];
250 		
251         switch (type) {
252 			case 'l': return toLisp(value);
253             case 's': return formatString(String(value), leftJustify, minWidth, precision, zeroPad, customPadChar);
254             case 'c': return formatString(String.fromCharCode(+value), leftJustify, minWidth, precision, zeroPad);
255             case 'b': return formatBaseX(value, 2, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
256             case 'o': return formatBaseX(value, 8, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
257             case 'x': return formatBaseX(value, 16, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
258             case 'X': return formatBaseX(value, 16, prefixBaseX, leftJustify, minWidth, precision, zeroPad).toUpperCase();
259             case 'u': return formatBaseX(value, 10, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
260             case 'i':
261             case 'd':
262                 number = parseInt(+value, 10);
263                 prefix = number < 0 ? '-' : positivePrefix;
264                 value = prefix + pad(String(Math.abs(number)), precision, '0', false);
265                 return justify(value, prefix, leftJustify, minWidth, zeroPad);
266             case 'e':
267             case 'E':
268             case 'f':
269             case 'F':
270             case 'g':
271             case 'G':
272                 number = +value;
273                 prefix = number < 0 ? '-' : positivePrefix;
274                 method = ['toExponential', 'toFixed', 'toPrecision']['efg'.indexOf(type.toLowerCase())];
275                 textTransform = ['toString', 'toUpperCase']['eEfFgG'.indexOf(type) % 2];
276                 value = prefix + Math.abs(number)[method](precision);
277                 return justify(value, prefix, leftJustify, minWidth, zeroPad)[textTransform]();
278             default: return substring;
279         }
280     };
281 
282     return format.replace(regex, doFormat);
283 }

287 function toJSON (object, pretty, levels, level) {
288 	levels = levels || 2; // Default levels
289 	level = level || 0;
290 	
291 	var done = level >= levels;
292 	var newline = pretty ? '\n' : '';
293 	var singleprefix = pretty ? times(' ', 2) : '';
294 	var prefix  = pretty ? times(singleprefix, level) : '';
295 	
296 	var json;
297 	var value;
298 	
299 	switch (typeof(object))
300 	{
301 	case 'undefined':
302 		return 'undefined';
303 	case 'object':
304 		if (!object) {
305 			// 'object' is null.
306 			return 'null';
307 		} else if (object instanceof Array) {
308 			// 'object' is an Array.
309 			json = '[';
310 			
311 			if (!done) {
312 				json += newline;
313 				value = null;
314 				for (var i = 0; i < object.length; i++) {
315 					value = toJSON(object[i], pretty, levels, level+1);
316 					json += prefix + singleprefix + value + ', ' + newline;
317 				}
318 				json = json.replace(/[\s\n]*$/g, '');
319 				json = json.replace(/,$/, '');
320 				if (object.length > 0) {
321 					json += newline + prefix;
322 				}
323 			} else {
324 				json += ' ... ';
325 			}
326 			
327 			return json + ']';
328 		} else if (object instanceof Date) {
329 			// 'object' is a Date.
330 			// Taken from http://www.json.org/json2.js
331 			function f (n) {
332 				// Format integers to have at least two digits.
333 				return n < 10 ? '0' + n : n;
334 			}
335 			
336 			return '"' +
337 				   object.getUTCFullYear()	   + '-' +
338 				   f(object.getUTCMonth() + 1) + '-' +
339 				   f(object.getUTCDate())	   + ' ' +
340 				   f(object.getUTCHours())	   + ':' +
341 				   f(object.getUTCMinutes())   + ':' +
342 				   f(object.getUTCSeconds())   +
343 				   '"';
344 		} else {
345 			json = '{';
346 			
347 			if (!done) {
348 				json = json + newline;
349 				count = 0;
350 				value = null;
351 				for (var key in object) {
352 					count++;
353 					if (object[key] == global) {
354 						value = "[global]";
355 					} else {
356 						value = toJSON(object[key], pretty, levels, level+1);
357 					}
358 					json += prefix + singleprefix + '"' + key + '": ' + value;
359 					json += ', ' + newline;
360 				}
361 				json = json.replace(/[\s\n]*$/g, '');
362 				json = json.replace(/,$/, '');
363 				if (count > 0) {
364 					json += newline + prefix;
365 				}
366 			} else {
367 				json += ' ... ';
368 			}
369 			
370 			return json + '}';
371 		}
372 	case 'function':
373 		var match = object.toString().match(/[^\(]+\([^\)]*\)/);
374 		return (match ? match[0] : 'function ()') + ' { ... }';
375 	case 'string':
376 		return '"' + object.replace(/"/g, '\\"') + '"';
377 	case 'number':
378 		return object;
379 	case 'boolean':	
380 	default:
381 		return object.toString();
382 	}
383 }

385 /**
386  * Returns a lispy string representation of the given value.
387  */
388 function toLisp (object) {
389 	if (object === null) {
390 		return "nil";
391 	}
392 	if (object === true) {
393 		return "t";
394 	}
395 	if (object === false) {
396 		return "f";
397 	}
398 	if (object instanceof Symbol) {
399 		return String(object);
400 	}
401 	if (object instanceof Keyword) {
402 		return ":" + String(object);
403 	}
404 	if (object instanceof Array) {
405 		return "(" + object.map(toLisp).join(" ") + ")";
406 	}
407 	return toJSON(object, true);
408 }

416 /**
417  * The method used for (equal) equality in js-lisp.
418  * 
419  * @returns Whether a and b are equal from js-lisp's perspective.
420  */
421 function equal (a, b) {
422 	// Test Symbol equality
423 	if (a instanceof Symbol) {
424 		if (!(b instanceof Symbol)) {
425 			return false;
426 		}
427 		return a.value == b.value;
428 	}
429 	
430 	// Test Keyword equality
431 	if (a instanceof Keyword) {
432 		if (!(b instanceof Keyword)) {
433 			return false;
434 		}
435 		return a.value == b.value;
436 	}
437 	
438 	// Test Array (list) equality
439 	if (a instanceof Array) {
440 		if (!(b instanceof Array)) {
441 			return false;
442 		}
443 		if (a.length !== b.length) {
444 			return false; // Return early in this easy and fast test
445 		}
446 		for (var i = 0; i < a.length; i++) {
447 			if (!equal(a[i], b[i])) {
448 				return false;
449 			}
450 		}
451 		return true;
452 	}
453 	
454 	return a == b;
455 }

465 function deepCopyArray (array) {
466 	var newArray = [];
467 	var item;
468 	for (var i = 0; i < array.length; i++) {
469 		item = array[i];
470 		if (item instanceof Array) {
471 			item = deepCopyArray(item);
472 		}
473 		newArray.push(item);
474 	}
475 	return newArray;

478 function makeRequest (url, successCallback) {
479 	var request;
480 	
481 	if (window.XMLHttpRequest) {
482 		request = new XMLHttpRequest();
483 	} else if (window.ActiveXObject) {
484 		request = new ActiveXObject("Msxml2.XMLHTTP");
485 	} else {
486 		throw new Error("Ajax request not supported in this browser");
487 	}
488 	
489 	request.open("GET", url, false); // Load the script synchronously
490 	request.send(null);
491 	
492 	if (request.status == 200) {
493 		successCallback(request.responseText);
494 	}
495 	
496 	assert(request.status != 404, "Trying to load lisp script that " +
497 		"does not exist: " + url);
498 }

923 function checkExplode (expression, parent, index) {
924 	if (expression instanceof Array) {
925 		if ((expression.length > 0) &&
926 			(equal(expression[0], _S("explode"))) &&
927 			parent) {
928 			var list = resolve(expression[1]);
929 			if (!(list instanceof Array)) {
930 				list = [list]; // Be lenient if someone is trying to "explode" a non-list
931 			}
932 			if (list.length === 0) {
933 				return index;
934 			}
935 			// Insert the expressions elements into the parent
936 			var end = parent.slice(index+1);
937 			parent.splice(index, 1); // Remove the (explode) expression
938 			parent.splice.apply(parent, [index, list.length].concat(list));
939 			parent.splice.apply(parent, [parent.length, end.length].concat(end));
940 			return index + list.length;
941 		} else {
942 			for (var i = 0; i < expression.length; i++) {
943 				i = checkExplode(expression[i], expression, i);
944 			}
945 		}
946 	}
947 	return parent ? index : expression;
948 }

950 function doSExp (sexp) {
951 	assert(!!sexp, "doSExp got empty expression");
952 	
953 	if (sexp.length === 0) {
954 		// An expression with no arguments, in js-lisp, is an empty list.
955 		return [];
956 	}
957 	
958 	var first = sexp[0];
959 	var object = resolve(first);
960 	
961 	assert(isCallable(object), "'" + first.value + "' is not callable");
962 	
963 	var thisObject = null;
964 	if (first instanceof Symbol) {
965 		var thisObjectPath = first.value.split(".").slice(0,-1).join(".");
966 		thisObject = lisp.env.get(thisObjectPath);
967 	}
968 	var args = sexp.slice(1);
969 	
970 	if (object instanceof lisp.Macro) {
971 		return object.callable.apply(thisObject, args);
972 	} else {
973 		return object.apply(thisObject, args.map(resolve));
974 	}
975 }

1004 var Env = Class.extend({
1005 	init: function (parent, symbols) {
1006 		this.parent = parent || null;
1007 		this.symbols = symbols || {};
1008 	},
1009 	
1010 	has: function (symbol) {
1011 		if (symbol instanceof Symbol) {
1012 			symbol = symbol.value;
1013 		}
1014 		
1015 		if (this.symbols.hasOwnProperty(symbol)) {
1016 			return true;
1017 		} else if (!this.parent) {
1018 			return false;
1019 		} else {
1020 			if (this.parent instanceof Env) {
1021 				return this.parent.has(symbol);
1022 			} else {
1023 				return this.parent[symbol] != undefined;
1024 			}
1025 		}
1026 	},
1027 	
1028 	get: function (symbol) {
1029 		symbol = String(symbol);
1030 		
1031 		var parts = symbol.split(".");
1032 		var value;
1033 		symbol = parts[0];
1034 		parts = parts.slice(1);
1035 		
1036 		if (this.symbols.hasOwnProperty(symbol) || this.symbols[symbol]) {
1037 			value = this.symbols[symbol];
1038 		} else if (!this.parent) {
1039 			value = undefined;
1040 		} else {
1041 			if (this.parent instanceof Env) {
1042 				value = this.parent.get(symbol);
1043 			} else {
1044 				value = this.parent[symbol];
1045 			}
1046 		}
1047 		
1048 		if (value && parts.length > 0) {
1049 			for (var i = 0; i < parts.length; i++) {
1050 				value = value[parts[i]];
1051 			}
1052 		}
1053 		
1054 		return value;
1055 	},
1056 	
1057 	set: function (symbol, value) {
1058 		symbol = String(symbol);
1059 		
1060 		var parts = symbol.split(".");
1061 		
1062 		if (parts.length > 1) {
1063 			var name = parts.slice(0,parts.length-1).join(".");
1064 			object = this.get(name);
1065 			
1066 			try {
1067 				object[parts[parts.length-1]] = value;
1068 			} catch (e) {
1069 				throw new Error(name + " is unsubscriptable: " + e);
1070 			}
1071 		} else {
1072 			if (this.has(symbol)) {
1073 				if (this.symbols.hasOwnProperty(symbol)) {
1074 					this.symbols[symbol] = value;
1075 				} else if (this.parent instanceof Env) {
1076 					try {
1077 						this.parent.set(symbol, value);
1078 					} catch (e) {
1079 						this.symbols[symbol] = value;
1080 					}
1081 				} else {
1082 					this.parent[symbol] = value;
1083 				}
1084 			} else {
1085 				var object = this;
1086 				while (object.parent && object.parent.symbols != global) {
1087 					object = object.parent;
1088 				}
1089 				object.symbols[symbol] = value;
1090 			}
1091 		}
1092 	},
1093 	
1094 	// FIXME: This method sucks.
1095 	let: function (symbol, value) {
1096 		symbol = String(symbol);
1097 		this.symbols[symbol] = value;
1098 	}

1228 /**

1230  */
1231 parse.any = function (stream) {
1232 	stream = validateInput(stream);
1233 	stream.swallowWhitespace();
1234

1236 	{
1237 	case '(':
1238 		return parse.sexp(stream);
1239 	case "'":
1240 		stream.next();
1241 		return [_S("quote"), parse.any(stream)];
1242 	case "`":
1243 		stream.next();
1244 		return [_S("backquote"), parse.any(stream)];
1245 	case "#":
1246 		stream.next();
1247 		switch (stream.peek())
1248 		{
1249 		case "'": // This is the function special operator
1250 			stream.next();
1251 			return [_S("function"), parse.any(stream)];
1252 		case "\\": // This is the char special operator
1253 			stream.next();
1254 			return [_S("char"), parse.symbol(stream)];

1256 		default:
1257 			// This is an unknown operator.

1259 			throw new parse.ParserException("Undefined special operator #" + c + " at " +
1260 				"position " + stream.position + " (expression: #" + c +
1261 				toLisp(parse.any(stream)) + ")");
1262 		}
1263 	case ",":
1264 		stream.next();

1271 	case ':':
1272 		return parse.keyword(stream);

1277 	default:
1278 		var rest = stream.rest();

1282 			if (match) {

1787  * valid uses of (catch) and (finally) in a (try) expression are:
1788  *

1792  *     - (try ... (catch ...) (finally ...))

1806  *     >> (try)

1808  * 
1809  * @example Empty catch block (silences the error)
1810  *     >> (try
1811  *            (throw (new Error))
1812  *          (catch))

1837  *            (throw (new Error))
1838  *          (catch (e)

1840  *          (finally
1841  *            (print "This always runs")))
