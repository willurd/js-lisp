 37 (function () {
 38 	var initializing = false;
 39 	var fnTest = (/xyz/).test(function(){xyz;}) ? (/\b_super\b/) : /.*/;
 40 	
 41 	/**
 42 	 * <p>Defines a base class from which to create new classes that can be
 43 	 * extended into new classes as well.</p>
 44 	 * 
 45 	 * <p>This constructor does nothing.</p>
 46 	 * 
 47 	 * <p>Modified from: http://ejohn.org/blog/simple-javascript-inheritance/</p>
 48 	 * 
 49 	 * @class
 50 	 * @name Class
 51 	 */
 52 	this.Class = function () {};
 53 	
 54 	/**
 55 	 * <p>Creates a new class that inherits from the calling class.</p>
 56 	 * 
 57 	 * @function
 58 	 * 
 59 	 * @param {string, object} classNameOrProps
 60 	 *     Either the name of the class, or an object containing the class' properties.
 61 	 * @param {object} props
 62 	 *     An object containing the class' properties (only if classNameOrProps
 63 	 *     specifies the class name).
 64 	 */
 65 	Class.extend = function (classNameOrProps, props) {
 66 		var _super = this.prototype;
 67 		var className = props ? classNameOrProps : "Class";
 68 		props = props || classNameOrProps;
 69 		
 70 		// Instantiate a base class (but only create the instance,
 71 		// don't run the init constructor)
 72 		initializing = true;
 73 		var prototype = new this();
 74 		initializing = false;
 75 		
 76 		// Copy the properties over onto the new prototype
 77 		for (var name in props) {
 78 			// Check if we're overwriting an existing function
 79 			if (typeof props[name] == "function" &&
 80 				typeof _super[name] == "function" &&
 81 				fnTest.test(props[name])) {
 82 				prototype[name] = (function(name, fn){
 83 					return function() {
 84 					var tmp = this._super;
 85 					
 86 					// Add a new ._super() method that is the same method
 87 					// but on the super-class
 88 					this._super = _super[name];
 89 					
 90 					// The method only need to be bound temporarily, so we
 91 					// remove it when we're done executing
 92 					var ret = fn.apply(this, arguments);        
 93 					this._super = tmp;
 94 					
 95 					return ret;
 96 					};
 97 				})(name, props[name]);
 98 			} else {
 99 				prototype[name] = props[name];
100 			}
101 		}
102 		
103 		// The new class
104 		var NewClass = function () {
105 			// All construction is actually done in the init method
106 			if (!initializing && this.init)
107 				this.init.apply(this, arguments);
108 		}
109 		
110 		NewClass.className = className;
111 		NewClass.prototype = prototype;
112 		NewClass.constructor = Class;
113 		NewClass.extend = arguments.callee;
114 		
115 		return NewClass;
116 	};

123 /**
124  * <p>Formats the string 'format' with the given arguments. Uses the
125  * php formatting style, defined at
126  * http://php.net/manual/en/function.sprintf.php.</p>
127  * 
128  * <p>From: http://phpjs.org/functions/sprintf:522.</p>
129  * 
130  * @param {string} format
131  *     The string to be formatted with the given arguments.
132  * @param {[mixed]} rest
133  *     The rest of the arguments with which to format the given string.
134  * 
135  * @example
136  *     >> sprintf("%01.2f", 123.1)
137  *     => 123.10
138  * @example
139  *     >> sprintf("[%10s]", 'monkey')
140  *     => "[    monkey]"
141  * @example
142  *     >> sprintf("[%'#10s]", 'monkey')
143  *     => "[####monkey]"
144  */
145 function sprintf (format /*, ... */) {
146     // http://kevin.vanzonneveld.net
147     // +   original by: Ash Searle (http://hexmen.com/blog/)
148     // + namespaced by: Michael White (http://getsprink.com)
149     // +    tweaked by: Jack
150     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
151     // +      input by: Paulo Ricardo F. Santos
152     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
153     // +      input by: Brett Zamir (http://brett-zamir.me)
154     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
155 
156     var regex = /%%|%(\d+\$)?([-+\'#0 ]*)(\*\d+\$|\*|\d+)?(\.(\*\d+\$|\*|\d+))?([scboxXuidfegEG])/g;
157     var a = arguments, i = 0, format = a[i++];
158 	
159     var pad = function (str, len, chr, leftJustify) {
160         if (!chr) {chr = ' ';}
161         var padding = (str.length >= len) ? '' : Array(1 + len - str.length >>> 0).join(chr);
162         return leftJustify ? str + padding : padding + str;
163     };
164 	
165     var justify = function (value, prefix, leftJustify, minWidth, zeroPad, customPadChar) {
166         var diff = minWidth - value.length;
167         if (diff > 0) {
168             if (leftJustify || !zeroPad) {
169                 value = pad(value, minWidth, customPadChar, leftJustify);
170             } else {
171                 value = value.slice(0, prefix.length) + pad('', diff, '0', true) + value.slice(prefix.length);
172             }
173         }
174         return value;
175     };
176 	
177     var formatBaseX = function (value, base, prefix, leftJustify, minWidth, precision, zeroPad) {
178         // Note: casts negative numbers to positive ones
179         var number = value >>> 0;
180         prefix = prefix && number && {'2': '0b', '8': '0', '16': '0x'}[base] || '';
181         value = prefix + pad(number.toString(base), precision || 0, '0', false);
182         return justify(value, prefix, leftJustify, minWidth, zeroPad);
183     };
184 
185     // formatString()
186     var formatString = function (value, leftJustify, minWidth, precision, zeroPad, customPadChar) {
187         if (precision != null) {
188             value = value.slice(0, precision);
189         }
190         return justify(value, '', leftJustify, minWidth, zeroPad, customPadChar);
191     };
192 
193     // doFormat()
194     var doFormat = function (substring, valueIndex, flags, minWidth, _, precision, type) {
195         var number;
196         var prefix;
197         var method;
198         var textTransform;
199         var value;
200 
201         if (substring == '%%') {return '%';}
202 
203         // parse flags
204         var leftJustify = false, positivePrefix = '', zeroPad = false, prefixBaseX = false, customPadChar = ' ';
205         var flagsl = flags.length;
206         for (var j = 0; flags && j < flagsl; j++) {
207             switch (flags.charAt(j)) {
208                 case ' ': positivePrefix = ' '; break;
209                 case '+': positivePrefix = '+'; break;
210                 case '-': leftJustify = true; break;
211                 case "'": customPadChar = flags.charAt(j+1); break;
212                 case '0': zeroPad = true; break;
213                 case '#': prefixBaseX = true; break;
214             }
215         }
216 
217         // parameters may be null, undefined, empty-string or real valued
218         // we want to ignore null, undefined and empty-string values
219         if (!minWidth) {
220             minWidth = 0;
221         } else if (minWidth == '*') {
222             minWidth = +a[i++];
223         } else if (minWidth.charAt(0) == '*') {
224             minWidth = +a[minWidth.slice(1, -1)];
225         } else {
226             minWidth = +minWidth;
227         }
228 
229         // Note: undocumented perl feature:
230         if (minWidth < 0) {
231             minWidth = -minWidth;
232             leftJustify = true;
233         }
234 
235         if (!isFinite(minWidth)) {
236             throw new Error('sprintf: (minimum-)width must be finite');
237         }
238 
239         if (!precision) {
240             precision = 'fFeE'.indexOf(type) > -1 ? 6 : (type == 'd') ? 0 : undefined;
241         } else if (precision == '*') {
242             precision = +a[i++];
243         } else if (precision.charAt(0) == '*') {
244             precision = +a[precision.slice(1, -1)];
245         } else {
246             precision = +precision;
247         }
248 		
249         // grab value using valueIndex if required?
250         value = valueIndex ? a[valueIndex.slice(0, -1)] : a[i++];
251 
252         switch (type) {
253             case 's': return formatString(String(value), leftJustify, minWidth, precision, zeroPad, customPadChar);
254             case 'c': return formatString(String.fromCharCode(+value), leftJustify, minWidth, precision, zeroPad);
255             case 'b': return formatBaseX(value, 2, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
256             case 'o': return formatBaseX(value, 8, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
257             case 'x': return formatBaseX(value, 16, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
258             case 'X': return formatBaseX(value, 16, prefixBaseX, leftJustify, minWidth, precision, zeroPad).toUpperCase();
259             case 'u': return formatBaseX(value, 10, prefixBaseX, leftJustify, minWidth, precision, zeroPad);
260             case 'i':
261             case 'd':
262                 number = parseInt(+value, 10);
263                 prefix = number < 0 ? '-' : positivePrefix;
264                 value = prefix + pad(String(Math.abs(number)), precision, '0', false);
265                 return justify(value, prefix, leftJustify, minWidth, zeroPad);
266             case 'e':
267             case 'E':
268             case 'f':
269             case 'F':
270             case 'g':
271             case 'G':
272                 number = +value;
273                 prefix = number < 0 ? '-' : positivePrefix;
274                 method = ['toExponential', 'toFixed', 'toPrecision']['efg'.indexOf(type.toLowerCase())];
275                 textTransform = ['toString', 'toUpperCase']['eEfFgG'.indexOf(type) % 2];
276                 value = prefix + Math.abs(number)[method](precision);
277                 return justify(value, prefix, leftJustify, minWidth, zeroPad)[textTransform]();
278             default: return substring;
279         }
280     };
281 
282     return format.replace(regex, doFormat);
283 }

287 function toJSON (object, pretty, levels, level) {
288 	levels = levels || 2; // Default levels
289 	level = level || 0;
290 	
291 	var done = level >= levels;
292 	var newline = pretty ? '\n' : '';
293 	var singleprefix = pretty ? times(' ', 2) : '';
294 	var prefix  = pretty ? times(singleprefix, level) : '';
295 	
296 	var json;
297 	var value;
298 	
299 	switch (typeof(object))
300 	{
301 	case 'undefined':
302 		return 'undefined';
303 	case 'object':
304 		if (!object) {
305 			// 'object' is null.
306 			return 'null';
307 		} else if (object instanceof Array) {
308 			// 'object' is an Array.
309 			json = '[';
310 			
311 			if (!done) {
312 				json += newline;
313 				value = null;
314 				for (var i = 0; i < object.length; i++) {
315 					value = toJSON(object[i], pretty, levels, level+1);
316 					json += prefix + singleprefix + value + ', ' + newline;
317 				}
318 				json = json.replace(/[\s\n]*$/g, '');
319 				json = json.replace(/,$/, '');
320 				if (object.length > 0) {
321 					json += newline + prefix;
322 				}
323 			} else {
324 				json += ' ... ';
325 			}
326 			
327 			return json + ']';
328 		} else if (object instanceof Date) {
329 			// 'object' is a Date.
330 			// Taken from http://www.json.org/json2.js
331 			function f (n) {
332 				// Format integers to have at least two digits.
333 				return n < 10 ? '0' + n : n;
334 			}
335 			
336 			return '"' +
337 				   object.getUTCFullYear()	   + '-' +
338 				   f(object.getUTCMonth() + 1) + '-' +
339 				   f(object.getUTCDate())	   + ' ' +
340 				   f(object.getUTCHours())	   + ':' +
341 				   f(object.getUTCMinutes())   + ':' +
342 				   f(object.getUTCSeconds())   +
343 				   '"';
344 		} else {
345 			json = '{';
346 			
347 			if (!done) {
348 				json = json + newline;
349 				count = 0;
350 				value = null;
351 				for (var key in object) {
352 					count++;
353 					if (object.hasOwnProperty(key)) {
354 						if (object[key] == global) {
355 							value = "[global]";
356 						} else {
357 							value = toJSON(object[key], pretty, levels, level+1);
358 						}
359 						json += prefix + singleprefix + '"' + key + '": ' + value;
360 						json += ', ' + newline;
361 					}
362 				}
363 				json = json.replace(/[\s\n]*$/g, '');
364 				json = json.replace(/,$/, '');
365 				if (count > 0) {
366 					json += newline + prefix;
367 				}
368 			} else {
369 				json += ' ... ';
370 			}
371 			
372 			return json + '}';
373 		}
374 	case 'function':
375 		var match = object.toString().match(/[^\(]+\([^\)]*\)/);
376 		return (match ? match[0] : 'function ()') + ' { ... }';
377 	case 'string':
378 		return '"' + object.replace(/"/g, '\\"') + '"';
379 	case 'number':
380 		return object;
381 	case 'boolean':	
382 	default:
383 		return object.toString();
384 	}
385 }

387 /**
388  * The method used for (equal) equality in js-lisp.
389  * 
390  * @returns Whether a and b are equal from js-lisp's perspective.
391  */
392 function equal (a, b) {
393 	// Test Symbol equality
394 	if (a instanceof Symbol) {
395 		if (!(b instanceof Symbol)) {
396 			return false;
397 		}
398 		return a.value == b.value;
399 	}
400 	
401 	// Test Keyword equality
402 	if (a instanceof Keyword) {
403 		if (!(b instanceof Keyword)) {
404 			return false;
405 		}
406 		return a.value == b.value;
407 	}
408 	
409 	// Test Array (list) equality
410 	if (a instanceof Array) {
411 		if (!(b instanceof Array)) {
412 			return false;
413 		}
414 		if (a.length !== b.length) {
415 			return false; // Return early in this easy and fast test
416 		}
417 		for (var i = 0; i < a.length; i++) {
418 			if (!equal(a[i], b[i])) {
419 				return false;
420 			}
421 		}
422 		return true;
423 	}
424 	
425 	return a == b;
426 }

484 var StringStream = Class.extend({
485 	init: function (data) {
486 		if (typeof(data) != "string") {
487 			throw new Error("Invalid object as StringStream input: " + data);
488 		}
489 
490 		this.data = data;
491 		this.length = data.length;
492 		this.position = 0;
493 		this.line = 1;
494 		this.column = 0;
495 	},
496 	
497 	slice: function () {
498 		return this.data.slice.apply(this.data, arguments);
499 	},
500 	
501 	rest: function (from) {
502 		from = from || this.position;
503 		return this.slice(from, this.data.length);
504 	},
505 	
506 	bof: function () {
507 		return this.position < 0;
508 	},
509 	
510 	eof: function () {
511 		return this.position >= this.length;
512 	},
513 	
514 	peek: function (distance) {
515 		distance = distance || 0;
516 		return this.charAt(this.position + distance);
517 	},
518 	
519 	charAt: function (index) {
520 		return this.data.charAt(index);
521 	},
522 	
523 	next: function () {
524 		if (this.eof()) {
525 			throw new StreamEOFException("EOF reached in StringStream");
526 		}
527 		
528 		var c = this.charAt(this.position);
529 		this.position += 1;
530 		this.column++;
531 		if (c == "\n") {
532 			this.line++;
533 			this.column = 0;
534 		}
535 		return c;
536 	},
537 	
538 	prev: function (count) {
539 		count = count || 1;
540 		this.position -= count;
541 
542 		if (this.bof()) {
543 			throw new Error("Cannot access character at position " + this.position +
544 				" of StringStream");
545 		}
546 		
547 		return this.charAt(this.position);
548 	},
549 	
550 	swallowWhitespace: function () {
551 		while (WHITESPACE.indexOf(this.peek()) != -1 && !this.eof()) {
552 			this.position++;
553 		}
554 	}

585 /**
586  * Recursively resolves all (resolve) expressions an
587  * unevaluated expression.
588  */
589 function checkResolve (expression) {
590 	if ((expression instanceof Array)) {
591 		if ((expression.length > 0) &&
592 			(equal(expression[0], _S("resolve")))) {
593 			return resolve(expression);
594 		}
595 		for (var i = 0; i < expression.length; i++) {
596 			expression[i] = checkResolve(expression[i]);
597 		}
598 	}
599 	return expression;

602 function doSExp (sexp) {
603 	if (!sexp) {
604 		throw new Error("doSExp got empty expression");
605 	}
606 	
607 	if (sexp.length === 0) {
608 		// An expression with no arguments, in js-lisp, is an empty list.
609 		return [];
610 	}
611 	
612 	var first = sexp[0];
613 	var object = resolve(first);
614 	
615 	if (typeof(object) != "function" && !(object instanceof Macro)) {
616 		throw new Error("'" + first.value + "' is not callable");
617 	}
618 	
619 	var thisObject = null;
620 	if (first instanceof Symbol) {
621 		var thisObjectPath = first.value.split(".").slice(0,-1).join(".");
622 		thisObject = lisp.env.get(thisObjectPath);
623 	}
624 	var args = sexp.slice(1);
625 	
626 	if (object instanceof lisp.Macro) {
627 		return object.callable.apply(thisObject, args);
628 	} else {
629 		return object.apply(thisObject, args.map(resolve));
630 	}
631 }

634 	if (args.length === 0) {
635 		return false;

639 			return false;
640 		}
641 	}

645 function comparator (args, testFunc) {
646 	if (args.length < 2) {
647 		return false;
648 	}
649 	var a = resolve(args[0]);
650 	for (var i = 1; i < args.length; i++) {
651 		var b = resolve(args[i]);
652 		if (!testFunc(a, b)) {
653 			return false;
654 		}
655 		a = b;
656 	}
657 	return true;

660 var Env = Class.extend({
661 	init: function (parent, symbols) {
662 		this.parent = parent || null;
663 		this.symbols = symbols || {};
664 	},
665 	
666 	has: function (symbol) {
667 		if (symbol instanceof Symbol) {
668 			symbol = symbol.value;
669 		}
670 		
671 		if (this.symbols.hasOwnProperty(symbol)) {
672 			return true;
673 		} else if (!this.parent) {
674 			return false;
675 		} else {
676 			if (this.parent instanceof Env) {
677 				return this.parent.has(symbol);
678 			} else {
679 				return this.parent[symbol] != undefined;
680 			}
681 		}
682 	},
683 	
684 	get: function (symbol) {
685 		symbol = String(symbol);
686 		
687 		var parts = symbol.split(".");
688 		var value;
689 		symbol = parts[0];
690 		parts = parts.slice(1);
691 		
692 		if (this.symbols.hasOwnProperty(symbol) || this.symbols[symbol]) {
693 			value = this.symbols[symbol];
694 		} else if (!this.parent) {
695 			value = undefined;
696 		} else {
697 			if (this.parent instanceof Env) {

1282 /**
1283  * <pre>
1284  * Provides JavaScript's try/catch/finally feature to the lisp
1285  * environment.
1286  * 
1287  * Note: (catch) and (finally) must be the last expressions in the
1288  * (try) expression, and (catch) must come before (finally). The only
1289  * valid uses of (catch) and (finally) in a (try) expression are:
1290  * 
1291  *     - (try ...) ; Not using either of them
1292  *     - (try ... (catch ...))
1293  *     - (try ... (finally ...))
1294  *     - (try ... (catch ...) (finally ...))
1295  * </pre>
1296  * 
1297  * @tested
1298  * 
1299  * @name try
1300  * @lisp
1301  * @function
1302  * @member lisp.macros
1303  * 
1304  * @returns The return value of the last evaluated body expression
1305  *          (i.e. non-catch, non-finally expression).
1306  * 
1307  * @example Empty try block
1308  *   >> (try)
1309  *   => nil
1310  * 
1311  * @example Empty catch block (silences the error)
1312  *   >> (try
1313  *          (throw (new Error))
1314  *        (catch))
1315  *   => nil
1316  * 
1317  * @example Multiple expressions with a full catch block
1318  *   >> (try
1319  *          (print "This will print")
1320  *          (throw (new Error "This cuts the expression short"))
1321  *          (print "This will not print")
1322  *        (catch (e)
1323  *          (format t "This will print when the error is thrown: %s" e)))
1324  *   => (no return value)
1325  * 
1326  * @example try/finally
1327  *   >> (try
1328  *          (throw (new Error))
1329  *        (finally
1330  *          (print "This always runs")))
1331  *   => (no return value) ; Due to the error that is thrown
1332  * 
1333  * @example try/catch/finally
1334  *   >> (try
1335  *          "hello" ; This will get returned because it's the last evaluated expression
1336  *          (throw (new Error))
1337  *        (catch (e)
1338  *          (print "This will get called"))
1339  *        (finally
1340  *          (print "This always runs")))
1341  *   => "hello" ; The error is caught, so there is a return value
1342  * 
1343  * @example catch/finally symbols as keywords
1344  *   >> (try
1345  *          (throw (new Error))
1346  *        (:catch) ; This will work
1347  *        (:finally)) ; And so will this
1348  *   => nil
1349  */
1350 defmacro("try", function () {
1351 	var args = argsToArray(arguments);
1352 	var checkIndex = args.length-1;
1353 	
1354 	// Grab the catch and finally expressions if they are there.
1355 	var check = function (symbol) {
1356 		if (checkIndex < 0) {
1357 			return null; // There are no expressions to check
1358 		}
1359 		var expr = args[checkIndex];
1360 		if ((expr instanceof Array) && // The "catch" expression must be a list
1361 			(expr.length > 0) && // It must at least have the symbol specifying the block type
1362 			((expr[0] instanceof Symbol) || (expr[0] instanceof Keyword)) &&
1363 			(expr[0].value == symbol)) {
1364 			args = args.slice(0, -1);
1365 			checkIndex--;
1366 			return expr;
1367 		}
1368 		return null;
1369 	};
1370 	var finallyExpression = check("finally");
1371 	var catchExpression = check("catch");
1372 	
1373 	var ret = null;
1374 	var i;
1375 	
1376 	try {
1377 		// Evaluate all of the expressions in the body, saving the return
1378 		// value of the last expression for returning from this expression.
1379 		for (i = 0; i < args.length; i++) {
1380 			ret = resolve(args[i]);
1381 		}
1382 	} catch (e) {
1383 		// Evaluate the `catch` expression if there is one.
1384 		if (catchExpression) {
1385 			var expression = [_S("lambda")].concat(catchExpression.slice(1));
1386 			if (expression.length === 1) { // Add an arglist if there isn't one
1387 				expression.push([]);
1388 			}
1389 			var callback = resolve(expression);
1390 			callback(e);
1391 		} else {
1392 			// If there is no catch expression, throw the error for something
1393 			// else to catch it (or not).
1394 			throw e;
1395 		}
1396 	} finally {
